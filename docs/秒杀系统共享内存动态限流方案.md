# 秒杀系统基于共享内存的动态限流方案

> Nginx `lua_shared_dict` + Redis 实现动态可调的分布式限流

## 一、现有静态限流的问题

### 1.1 当前配置

```nginx
# 静态限流（固定速率）
limit_req_zone $binary_remote_addr zone=flash_ip:10m rate=3r/s;
limit_req_zone $http_x_user_id zone=flash_user:10m rate=1000r/s;
limit_req_zone $request_uri zone=flash_grab:10m rate=5000r/s;
```

### 1.2 问题分析

| 问题 | 影响 |
|---|---|
| **改限流必须 reload** | 修改 `rate` 需要 `nginx -s reload`，影响现有连接 |
| **不能自适应** | 秒杀前期流量大、后期流量小，固定速率浪费资源 |
| **单机隔离** | 每台 Nginx 独立计数，无法感知全局流量 |
| **无动态优先级** | VIP 用户和普通用户共享限流阈值 |

## 二、基于共享内存的动态限流架构

```
                          ┌─────────────────────┐
                          │    Redis 全局限流    │
                          │  (跨节点共享计数)     │
                          └──────┬──────────────┘
                                 │ 定期同步限流配置
                                 ▼
┌────────────────────────────────────────────────────┐
│                   Nginx 节点 1                       │
│  ┌──────────────────────────────────────────────┐  │
│  │  lua_shared_dict flash_limit 100M             │  │
│  │  lua_shared_dict flash_config 10M             │  │
│  │                                               │  │
│  │  Worker 1 (计数器)   Worker 2 (计数器)         │  │
│  │  Worker 3 (计数器)   Worker 4 (计数器)         │  │
│  └──────────────────────────────────────────────┘  │
└────────────────────────────────────────────────────┘

限流维度（三层漏斗）:

用户请求
    │
    ▼
┌──────────────┐
│  ① 本地共享内存 │  ← 单机维度：毫秒级响应，无网络开销
│  漏桶/令牌桶    │     维度：IP + User + URI
└──────┬───────┘
       │ 超过本地阈值
       ▼
┌──────────────┐
│  ② Redis 集群  │  ← 全局维度：秒级同步，跨节点共享
│  滑动窗口      │     维度：IP + User + SKU
└──────┬───────┘
       │ 超过全局阈值
       ▼
┌──────────────┐
│  ③ 后端保护    │  ← 后端压力感知：自适应降速
│  自适应降级    │     根据后端响应时间动态调整
└──────────────┘
```

### 3.2 动态限流核心模块

```lua
-- /etc/nginx/lua/flash_ratelimit.lua
--
-- 基于共享内存的动态限流模块
-- 支持运行时调整限流参数，无需 reload nginx

local ngx       = ngx
local shared    = ngx.shared
local time      = ngx.time
local log       = ngx.log
local WARN      = ngx.WARN
local ERR       = ngx.ERR
local DEBUG     = ngx.DEBUG

-- ======== 配置常量（默认值，可从 flash_config 覆盖） ========
local DEFAULTS = {
    user_qps       = 1000,    -- 每用户 QPS
    user_burst     = 200,     -- 用户突发
    ip_qps         = 3,       -- 每 IP QPS
    ip_burst       = 5,       -- IP 突发
    total_qps      = 5000,    -- 总入口 QPS
    total_burst    = 1000,    -- 总突发
    concurrency    = 5,       -- 每 IP 并发
    -- 动态调节参数
    downgrade_rate = 1.0,     -- 降级系数 (0.0~1.0)
    enable_vip     = true,    -- VIP 用户免限流
}

local _M = {}

-- ======== 从共享内存获取动态配置 ========
function _M.get_config(key)
    local val = shared.flash_config:get("config:" .. key)
    if val then
        return val
    end
    return DEFAULTS[key]
end

-- ======== 从共享内存获取数值配置 ========
function _M.get_number(key)
    local val = _M.get_config(key)
    if type(val) == "number" then
        return val
    end
    return tonumber(val) or DEFAULTS[key]
end

-- ======== 判断 VIP 用户 ========
function _M.is_vip(uid)
    if not uid then return false end
    -- 从共享内存中的 VIP 列表查询（由管理 API 更新）
    local vips = shared.flash_config:get("vip_list")
    if vips then
        local vip_set = cjson.decode(vips)
        if vip_set and vip_set[uid] then
            return true
        end
    end
    return false
end

-- ======== IP 级本地限流（漏桶） ========
--
-- 使用 lua_shared_dict 实现跨 worker 计数
-- 每个 IP 一个计数器，过期时间 1 秒
--
-- 返回: true=放行, false=限流
function _M.check_ip_limit(ip)
    if not ip then return true end

    local ip_qps = _M.get_number("ip_qps")
    local ip_burst = _M.get_number("ip_burst")
    local downgrade = _M.get_number("downgrade_rate")

    -- 应用降级系数
    local effective_qps = math.max(1, math.floor(ip_qps * downgrade))

    local key = "ip:" .. ip
    local current = shared.flash_local_cnt:get(key) or 0
    local window_ts = math.floor(time() / 1)  -- 1 秒窗口

    -- 使用时间戳作为 key 的一部分实现滑动窗口
    local window_key = key .. ":" .. window_ts
    local count, err = shared.flash_local_cnt:incr(window_key, 1, 0, 2)  -- TTL 2 秒

    if count then
        if count > effective_qps + ip_burst then
            log(WARN, "IP限流: ", ip, " count=", count, " limit=", effective_qps)
            return false
        end
    end

    return true
end

-- ======== 用户级限流（令牌桶） ========
--
-- 基于令牌桶算法，支持突发
-- 桶容量 = burst，每秒补充 QPS 个令牌
--
-- 返回: true=放行, false=限流
function _M.check_user_limit(uid)
    if not uid then return true end

    -- VIP 用户免限流
    if _M.get_config("enable_vip") and _M.is_vip(uid) then
        return true
    end

    local user_qps = _M.get_number("user_qps")
    local user_burst = _M.get_number("user_burst")
    local downgrade = _M.get_number("downgrade_rate")

    local effective_qps = math.max(1, math.floor(user_qps * downgrade))

    local key = "user:" .. uid
    local window_key = key .. ":" .. math.floor(time() / 1)
    local count = shared.flash_local_cnt:get(window_key) or 0

    if count >= effective_qps + user_burst then
        log(DEBUG, "用户限流: ", uid, " count=", count)
        return false
    end

    shared.flash_local_cnt:incr(window_key, 1, 0, 2)
    return true
end

-- ======== 总入口限流 ========
function _M.check_total_limit()
    local total_qps = _M.get_number("total_qps")
    local total_burst = _M.get_number("total_burst")
    local downgrade = _M.get_number("downgrade_rate")

    local effective_qps = math.max(1, math.floor(total_qps * downgrade))
    local key = "total:" .. math.floor(time() / 1)

    local count = shared.flash_local_cnt:incr(key, 1, 0, 2)
    if count and count > effective_qps + total_burst then
        log(WARN, "总入口限流: count=", count, " limit=", effective_qps)
        return false
    end

    return true
end

-- ======== 三重检查（合并调用） ========
--
-- IP → 用户 → 总入口 逐级检查
-- 任意一级失败则限流
--
-- 返回: true=放行, false=限流
function _M.check_all(ip, uid)
    -- 1. IP 级检查
    if not _M.check_ip_limit(ip) then
        return false, "IP_REQUEST_LIMIT"
    end

    -- 2. 用户级检查
    if not _M.check_user_limit(uid) then
        return false, "USER_REQUEST_LIMIT"
    end

    -- 3. 总入口检查
    if not _M.check_total_limit() then
        return false, "TOTAL_REQUEST_LIMIT"
    end

    return true, "OK"
end

-- ======== 实时调整限流参数 ========
--
-- 从管理 API 或 Redis 同步最新的限流配置
-- 写入 lua_shared_dict，worker 即时生效
--
function _M.update_config(key, value)
    local ok, err = shared.flash_config:set("config:" .. key, value)
    if ok then
        log(WARN, "限流配置更新: ", key, "=", tostring(value))
    else
        log(ERR, "限流配置更新失败: ", err)
    end
    return ok
end

-- ======== 批量更新配置（从 Redis 同步） ========
function _M.sync_from_redis(redis_host, redis_port)
    local redis = require("resty.redis")
    local red = redis:new()
    red:set_timeout(100)  -- 100ms 超时

    local ok, err = red:connect(redis_host, redis_port)
    if not ok then
        log(ERR, "连接 Redis 失败: ", err)
        return
    end

    -- 读取动态限流配置
    local config_keys = red:keys("nginx:ratelimit:*")
    if config_keys then
        for _, key in ipairs(config_keys) do
            local val = red:get(key)
            if val then
                local config_name = key:match("nginx:ratelimit:(.*)")
                _M.update_config(config_name, val)
            end
        end
    end

    -- 读取 VIP 名单
    local vip_data = red:get("nginx:vip_list")
    if vip_data then
        shared.flash_config:set("vip_list", vip_data)
    end

    red:close()
end

return _M
```

## 四、Redis 协同全局限流

### 4.1 Redis 计数（跨节点共享）

当本地共享内存统计到超出阈值时，再查 Redis 全局计数：

```lua
-- /etc/nginx/lua/global_limit.lua
--
-- 全局限流：跨 Nginx 节点共享计数器
-- 基于 Redis 滑动窗口（精确到秒）
-- 只在本地限流放行后才检查，减少 Redis 压力

local redis = require("resty.redis")
local shared = ngx.shared

local REDIS_HOST = "hadoop003"
local REDIS_PORT = 6379
local REDIS_TIMEOUT = 50  -- 50ms 超时

local _M = {}

-- ======== 获取 Redis 连接（连接池复用） ========
local function get_redis()
    local red = redis:new()
    red:set_timeout(REDIS_TIMEOUT)
    local ok, err = red:connect(REDIS_HOST, REDIS_PORT)
    if not ok then
        ngx.log(ngx.ERR, "Redis 连接失败: ", err)
        return nil
    end
    -- 使用连接池
    red:set_keepalive(30000, 100)  -- 30 秒空闲，100 个连接
    return red
end

-- ======== 全局 IP 限流检查 ========
--
-- Redis Key: flash:global:ip:{ip}:{秒级时间戳}
-- 仅在本地限流通过后调用
--
function _M.check_global_ip(ip)
    local red = get_redis()
    if not red then return true end  -- Redis 不可用则放行

    local key = "flash:global:ip:" .. ip .. ":" .. math.floor(ngx.time() / 1)
    local count, err = red:incr(key)
    if count then
        red:expire(key, 5)  -- 5 秒过期
        if count > 10 then  -- 全局每个 IP 每秒最多 10 次
            ngx.log(ngx.WARN, "全局IP限流: ", ip, " count=", count)
            return false
        end
    end
    return true
end

-- ======== 全局 SKU 限流检查 ========
--
-- Redis Key: flash:global:sku:{fsId}:{秒级时间戳}
--
function _M.check_global_sku(fsId)
    if not fsId then return true end

    local red = get_redis()
    if not red then return true end

    local key = "flash:global:sku:" .. fsId .. ":" .. math.floor(ngx.time() / 1)
    local count, err = red:incr(key)
    if count then
        red:expire(key, 5)
        if count > 10000 then  -- 全局每个 SKU 每秒最多 10000
            ngx.log(ngx.WARN, "全局SKU限流: ", fsId, " count=", count)
            return false
        end
    end
    return true
end

return _M
```

### 5.1 Nginx 内部管理端点

```nginx
# 限流配置管理接口（只允许内网访问）
server {
    listen 127.0.0.1:81;

    location = /ratelimit/update {
        content_by_lua_block {
            local cjson = require("cjson")
            local flash_ratelimit = require("flash_ratelimit")

            ngx.req.read_body()
            local body = ngx.req.get_body_data()
            if not body then
                ngx.say('{"code":-1,"msg":"body required"}')
                return
            end

            local ok, data = pcall(cjson.decode, body)
            if not ok then
                ngx.say('{"code":-1,"msg":"invalid json"}')
                return
            end

            for key, value in pairs(data) do
                flash_ratelimit.update_config(key, value)
            end

            ngx.say('{"code":0,"msg":"ok"}')
        }
    }

    location = /ratelimit/status {
        content_by_lua_block {
            local cjson = require("cjson")
            local shared = ngx.shared

            local status = {
                config = {
                    user_qps    = shared.flash_config:get("config:user_qps"),
                    ip_qps      = shared.flash_config:get("config:ip_qps"),
                    total_qps   = shared.flash_config:get("config:total_qps"),
                    downgrade_rate = shared.flash_config:get("config:downgrade_rate"),
                },
                stats = {
                    local_cnt_capacity = shared.flash_local_cnt:capacity(),
                    local_cnt_free     = shared.flash_local_cnt:free_space(),
                }
            }

}
```

### 5.2 外部控制台配置推送

```java
// NginxRatelimitController.java — 后端管理接口
@RestController
@RequestMapping("/admin/nginx/ratelimit")
public class NginxRatelimitController {

    @Autowired
    private StringRedisTemplate redisTemplate;

    /**
     * 动态调整 Nginx 限流参数
     *
     * 将配置写入 Redis，Nginx worker 进程
     * 每 5 秒从 Redis 同步，无需 reload
     */
    @PostMapping("/update")
    public ApiResult<String> updateRatelimit(@RequestBody Map<String, Object> config) {
        for (Map.Entry<String, Object> entry : config.entrySet()) {
            redisTemplate.opsForValue().set(
                "nginx:ratelimit:" + entry.getKey(),
                String.valueOf(entry.getValue()));
        }
        // 立即通知所有 Nginx 节点同步
        redisTemplate.convertAndSend("nginx:ratelimit:sync", "refresh");
        return ApiResult.success("配置已更新，等待 Nginx 同步");
    }

    /**
     * 秒杀活动开始时自动调优
     */
    @PostMapping("/flash/start")
    public ApiResult<String> onFlashStart(@RequestParam String fsId) {
        Map<String, Object> flashConfig = new HashMap<>();
        flashConfig.put("user_qps", 2000);       // 放宽用户限流
        flashConfig.put("total_qps", 10000);     // 放宽总入口
        flashConfig.put("downgrade_rate", 1.0);  // 恢复正常
        return updateRatelimit(flashConfig);
    }

    /**
     * 秒杀活动结束时恢复默认
     */
    @PostMapping("/flash/end")
    public ApiResult<String> onFlashEnd(@RequestParam String fsId) {
        Map<String, Object> defaultConfig = new HashMap<>();
        defaultConfig.put("user_qps", 1000);
        defaultConfig.put("total_qps", 5000);
        defaultConfig.put("downgrade_rate", 1.0);
        return updateRatelimit(defaultConfig);
    }
}
```

## 六、自适应降级

### 6.1 根据后端压力动态调整

```lua
-- /etc/nginx/lua/auto_downgrade.lua
--
-- 自动降级：根据后端响应时间和错误率动态调整限流阈值
-- 写入 flash_config 共享内存，即时生效

local shared = ngx.shared
local time = ngx.time
local log = ngx.log
local WARN = ngx.WARN
local ERR = ngx.ERR

local _M = {}

-- 滑动窗口统计
local stats_window = {}
local WINDOW_SIZE = 10  -- 统计最近 10 秒

-- ======== 记录后端响应 ========
function _M.record_upstream(time_ms, status)
    local now = time()
    local second = math.floor(now)

    if not stats_window[second] then
        stats_window[second] = { total = 0, slow = 0, error = 0, sum_ms = 0 }
        -- 清理旧数据
        for s in pairs(stats_window) do
            if now - s > 60 then
                stats_window[s] = nil
            end
        end
    end

    local stat = stats_window[second]
    stat.total = stat.total + 1
    stat.sum_ms = stat.sum_ms + time_ms

    if time_ms > 1000 then  -- > 1 秒算慢请求
        stat.slow = stat.slow + 1
    end
    if status >= 500 then
        stat.error = stat.error + 1
    end
end

-- ======== 计算降级系数 ========
--
-- 根据最近 10 秒的统计数据自动计算降级系数
-- 返回: 0.0 ~ 1.0
--
function _M.calculate_downgrade_rate()
    local now = time()
    local total, slow, err = 0, 0, 0

    for s, stat in pairs(stats_window) do
        if now - s <= WINDOW_SIZE then
            total = total + stat.total
            slow = slow + stat.slow
            err = err + stat.error
        end
    end

    if total == 0 then return 1.0 end

    local slow_rate = slow / total
    local err_rate = err / total
    local weighted = slow_rate * 0.4 + err_rate * 0.6  -- 加权评分

    -- 降级系数 = 1.0 - 加权评分
    local rate = math.max(0.1, math.min(1.0, 1.0 - weighted))

    if rate < 1.0 then
        log(WARN, "自适应降级: rate=", rate, " slow=", slow_rate, " err=", err_rate)
    end

    return rate
end

-- ======== 每 3 秒更新一次降级系数 ========
function _M.start_auto_downgrade()
    local flash_ratelimit = require("flash_ratelimit")
    local ngx_timer = ngx.timer

    local function adjust(premature)
        if premature then return end
        local rate = _M.calculate_downgrade_rate()
        flash_ratelimit.update_config("downgrade_rate", rate)
        ngx_timer.every(3, adjust)
    end

    ngx_timer.every(3, adjust)
end

return _M
```

```nginx
    -- 记录后端响应时间（用于自适应降级）
    body_filter_by_lua_block {
        local auto_downgrade = require("auto_downgrade")
        local request_time = ngx.var.request_time  -- 单位: 秒
        auto_downgrade.record_upstream(
            request_time * 1000,   -- 转为毫秒
            ngx.status
        )
    }
}
```

## 七、VIP 用户免限流

### 7.1 VIP 名单管理

```java
// VipListService.java — VIP 管理
@Service
public class VipListService {

    @Autowired
    private StringRedisTemplate redisTemplate;

    private static final String KEY = "nginx:vip_list";

    /**
     * 添加 VIP 用户
     */
    public void addVip(String uid) {
        redisTemplate.opsForHash().put(KEY, uid, "1");
        // 通知 Nginx 刷新
        redisTemplate.convertAndSend("nginx:ratelimit:sync", "vip_update");
    }

    /**
     * 批量添加 VIP（秒杀活动运营人员）
     */
    public void batchAddVip(List<String> uids) {
        Map<String, String> map = new HashMap<>();
        uids.forEach(uid -> map.put(uid, "1"));
        redisTemplate.opsForHash().putAll(KEY, map);
    }

    /**
     * 移除 VIP
     */
    public void removeVip(String uid) {
        redisTemplate.opsForHash().delete(KEY, uid);
    }
}
```

## 八、与静态限流的对比

| 维度 | 静态 `limit_req_zone` | 动态共享内存限流 |
|---|---|---|
| **调整限流** | 改配置→reload Nginx→中断连接 | API 写入 Redis→5 秒内生效→零中断 |
| **统计粒度** | IP、URI 级别 | IP、用户、SKU、VIP 任意维度 |
| **跨节点共享** | 否，单机独立计数 | 是，Redis 全局计数 |
| **自适应** | 否，固定速率 | 是，根据后端压力自动降级 |
| **VIP 支持** | 不支持 | 支持 VIP 白名单免限流 |
| **性能** | 内核态，极高 | Lua 用户态，略低于内核限流 |
| **复杂度** | 低，一行配置 | 中，需 Lua 脚本+Redis |
| **回滚** | 改配置→reload | API 恢复默认值→秒级生效 |

### 9.2 秒杀活动自动调优

```java
// FlashSaleLifecycleListener.java
@Component
public class FlashSaleLifecycleListener {

    @Autowired
    private StringRedisTemplate redisTemplate;

    /**
     * 秒杀活动开始 → 放宽限流
     */
    public void onFlashStart(String fsId) {
        Map<String, String> config = Map.of(
            "user_qps", "2000",
            "total_qps", "10000",
            "downgrade_rate", "1.0"
        );
        config.forEach((k, v) ->
            redisTemplate.opsForValue().set("nginx:ratelimit:" + k, v));
        redisTemplate.convertAndSend("nginx:ratelimit:sync", "flash_start");
    }

    /**
     * 秒杀活动结束 → 恢复默认
     */
    public void onFlashEnd(String fsId) {
        Map<String, String> config = Map.of(
            "user_qps", "1000",
            "total_qps", "5000",
            "downgrade_rate", "0.5"
        );
        config.forEach((k, v) ->
            redisTemplate.opsForValue().set("nginx:ratelimit:" + k, v));
        redisTemplate.convertAndSend("nginx:ratelimit:sync", "flash_end");
    }
}
```
